# 二、后端搭建（Node + Express）
## 1. 初始化项目
```bash
mkdir rent-server
cd rent-server
npm init -y
npm install express mysql2 cors
```

## 2. 入口文件：index.js（完整可运行）
```js
const express = require('express');
const mysql = require('mysql2/promise');
const cors = require('cors');
const app = express();

// 中间件
app.use(cors());
app.use(express.json());

// 数据库连接池
const pool = mysql.createPool({
  host: 'localhost',
  user: 'root',
  password: '你的数据库密码',
  database: 'rent_system',
  waitForConnections: true,
  connectionLimit: 10,
  queueLimit: 0
});

// 测试接口
app.get('/', (req, res) => {
  res.send('房东收租管理系统 - 后端服务运行中');
});

// ======================================
// 用户模块（登录/注册）
// ======================================
// 注册
app.post('/api/register', async (req, res) => {
  const { username, password } = req.body;
  try {
    await pool.query(
      'INSERT INTO sys_user (username, password) VALUES (?, ?)',
      [username, password]
    );
    res.json({ code: 200, msg: '注册成功' });
  } catch (err) {
    res.json({ code: 500, msg: '注册失败', err });
  }
});

// 登录
app.post('/api/login', async (req, res) => {
  const { username, password } = req.body;
  const [rows] = await pool.query(
    'SELECT id, username, nickname FROM sys_user WHERE username=? AND password=?',
    [username, password]
  );
  if (rows.length === 0) {
    return res.json({ code: 401, msg: '账号或密码错误' });
  }
  res.json({ code: 200, msg: '登录成功', data: rows[0] });
});

// ======================================
// 房源模块
// ======================================
// 获取房源列表
app.get('/api/house/list', async (req, res) => {
  const { user_id } = req.query;
  const [rows] = await pool.query(
    'SELECT * FROM house_info WHERE user_id=?',
    [user_id]
  );
  res.json({ code: 200, data: rows });
});

// 添加房源
app.post('/api/house/add', async (req, res) => {
  const { user_id, community_name, house_no, rent_price, water_price, electric_price } = req.body;
  await pool.query(
    'INSERT INTO house_info (user_id, community_name, house_no, rent_price, water_price, electric_price) VALUES (?,?,?,?,?,?)',
    [user_id, community_name, house_no, rent_price, water_price, electric_price]
  );
  res.json({ code: 200, msg: '添加成功' });
});

// 启动服务
const PORT = 3000;
app.listen(PORT, () => {
  console.log(`后端运行：http://localhost:${PORT}`);
});
```

# 三、前端搭建（Vue3 + Element Plus）
## 1. 创建项目
```bash
npm create vite@latest rent-web -- --template vue
```

## 2. main.js 配置
```js
import { createApp } from 'vue'
import ElementPlus from 'element-plus'
import 'element-plus/dist/index.css'
import App from './App.vue'
import axios from 'axios'

const app = createApp(App)
app.use(ElementPlus)
app.config.globalProperties.$axios = axios
app.mount('#app')
```

## 3. axios 配置（src/utils/request.js）
```js
import axios from 'axios'
const request = axios.create({
  baseURL: 'http://localhost:3000/api'
})

```

## 4. 登录页面（src/views/Login.vue）
```vue
<template>
  <div class="login-box">
    <el-card class="login-card">
      <h2>房东收租管理系统</h2>
      <el-form @submit.prevent="login">
        <el-form-item>
          <el-input v-model="form.username" placeholder="账号" />
        </el-form-item>
        <el-form-item>
          <el-input v-model="form.password" placeholder="密码" type="password" />
        </el-form-item>
        <el-button type="primary" round block @click="login">登录</el-button>
      </el-form>
    </el-card>
  </div>
</template>

<script setup>
import { ref } from 'vue'
import request from '../utils/request'
import { ElMessage } from 'element-plus'
import { useRouter } from 'vue-router'
const router = useRouter()
const form = ref({ username: '', password: '' })

const login = async () => {
  const res = await request.post('/login', form.value)
  if (res.data.code === 200) {
    ElMessage.success('登录成功')
    localStorage.setItem('user', JSON.stringify(res.data.data))
    router.push('/dashboard')
  } else {
    ElMessage.error(res.data.msg)
  }
}
</script>

<style>
.login-box{display:flex;justify-content:center;align-items:center;height:100vh;background:#f5f5f5}
.login-card{width:400px;padding:20px}
</style>
```
